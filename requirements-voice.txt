# Local voice control (Stage C of docs-grocery-voice-plan.md) — OPTIONAL.
# Install only on a box with a working mic + audio libs:
#     sudo apt install -y libportaudio2
#     .venv/bin/pip install -r requirements-voice.txt
#
# openwakeword >=0.6 hard-requires tflite-runtime, which has no wheel for many
# Python/CPU combos (incl. the wall's Python 3.12). Pinning <0.6 + onnxruntime
# uses the ONNX inference path instead, which has wheels everywhere. These are
# split out of the core requirements.txt so the grocery + photo features always
# install cleanly even when the voice stack doesn't.
#
# Hearing (Deepgram Nova) and the voice (Deepgram Aura) run in the cloud over
# plain HTTPS — no SDK needed. faster-whisper + piper-tts below are now only the
# *local fallbacks* used when DEEPGRAM_API_KEY is unset or a call fails.
onnxruntime>=1.17       # wake word (always local)
openwakeword<0.6        # wake word (always local)
sounddevice>=0.4        # mic capture (always local)
webrtcvad-wheels>=2.0   # end-of-speech detection; prebuilt wheels (plain webrtcvad
                        # needs a C toolchain the wall lacks). Imports as `webrtcvad`.
faster-whisper>=1.0     # local STT fallback (cloud is primary)
piper-tts>=1.2          # local TTS fallback (cloud is primary)
